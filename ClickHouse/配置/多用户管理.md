在/etc/clickhouse-server/users.d下创建用户文件即可：

```xml
<?xml version="1.0"?>
<yandex>
    <users>
        <readset>
            <!--PASSWORD=$(base64 < /dev/urandom | head -c12); echo "$PASSWORD";-->
            <password>qqBvbBGyX+ii</password>
            <networks incl="networks" replace="replace">
                <ip>::/0</ip>
            </networks>
            <profile>readset</profile>
            <quota>default</quota>
        </readset>
    </users>
</yandex>
```
